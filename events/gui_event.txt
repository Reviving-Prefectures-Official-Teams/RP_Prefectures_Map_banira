namespace = rp_gui

rp_gui.1 = { # 建設局を首都に自動設置
	type = country_event

	duration = 3

	trigger = {
		THIS = c:KNG
		always = yes
		# capital = {
		# 	NOT = {
		# 		has_building = "building_construction_sector_rp"
		# 	}
		# }
	}

	immediate = {
		capital = {
			create_building = {
				building = "building_construction_sector_rp"
				level=1
				reserves=1
			}
		}
	}

	option = {
		name = rp_gui.1.a
	}
}

rp_gui.2 = { # 首都以外の建設局を削除
	type = country_event
	placement = root

	duration = 0

	trigger = {
		any_state = {
			is_capital = no
			has_building = "building_construction_sector_rp"
		}
	}

	immediate = {
		every_state = {
			limit = {
				is_capital = no
				has_building = "building_construction_sector_rp"
			}
			remove_building = "building_construction_sector_rp"
		}
	}

	option = {
		name = rp_gui.2.a
	}
}
