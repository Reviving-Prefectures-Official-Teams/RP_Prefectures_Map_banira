                            OR = {

                                nordic_clothes_pop_trigger = yes

                    }
                    african_diaspora_pop_clothes_trigger = yes
                    AND = {
                        native_american_clothes_pop_trigger = yes

                            scope:pop ?= {
                                NOT = { culture = cu:inuit }
                            }
                        }
                    }
                }

                scope:pop ?= {
                    NAND = {
                        religion = rel:catholic
                        month = 1 # Carnival is /usually/ in February
                    }
                }
            }
        }
