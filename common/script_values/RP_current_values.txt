investment_pool_spending = { 
	value = investment_pool_gross_income
	subtract = investment_pool_net_income
}

#投資プール支出は、投資プール総収入 - 投資プール純収入

construction_price = {
	value = this.market.mg:construction.market_goods_pricier
	add = 1
	multiply = 130
}

#建設価格は、建設市場の価格が高い + 1 * 130

private_construction_bought = {
	value = investment_pool_spending
	divide = construction_price
}

#民間建設購入は、投資プール支出 / 建設価格

investment_price_incentive = {
	value = capital.sg:construction.state_goods_cheaper
	add = 2.25
	divide = 2.25
}

#投資価格インセンティブは、首都の建設国家財が安い + 2.25 / 2.25

total_construction_use = {
	value = capital.sg:construction.state_goods_consumption
	divide = capital.modifier:state_market_access_price_impact
}

#建設総使用量は、首都の建設国家財消費 / 首都の市場アクセス価格影響

total_construction_spending = {
	value = total_construction_use
	multiply = construction_price
}

#建設総支出は、建設総使用量 * 建設価格

public_construction_spending = {
	value = total_construction_spending
	subtract = investment_pool_spending
}

#公共建設支出は、建設総支出 - 投資プール支出

private_construction_use = {
	value = investment_pool_spending
	divide = construction_price
}

#民間建設使用量は、投資プール支出 / 建設価格

public_construction_use = {
	value = public_construction_spending
	divide = construction_price
}

#公共建設使用量は、公共建設支出 / 建設価格
