RP_buildings_under_construction = {
	value = 0
	every_scope_building = {
		limit = {
			is_under_construction = yes
		}
		add = 1
	}
}

#建設中の建物数
##基本の値は0
##建設中の建物がある場合、1を加算

RP_amount_of_sectors = {
	value = 0
	every_scope_building = {
		limit = {
			type = bt:building_construction_sector
		}
		add = 1
	}
}

#セクターの数
##基本の値は0
##建設セクターの場合、1を加算

RP_construction_per_sector = {
	value = total_construction_use_target
	if = {
		limit = {
			RP_amount_of_sectors > 0
		}
		divide = RP_amount_of_sectors
	}
	#subtract = 1
}

#セクターごとの建設
##基本の値は建設目標値
##セクターの数が0より大きい場合、セクターの数で割る
