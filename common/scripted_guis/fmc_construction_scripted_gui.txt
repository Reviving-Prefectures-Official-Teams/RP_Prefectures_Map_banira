add_construction_scriptedgui = {

	scope = country 

	is_shown = {
        always = yes
    } 
    
	ai_is_valid = {
        always = no
    }

	is_valid = {
        always = yes
    } 
    
    effect = {	
		#custom_tooltip = ""
        initialize_gov_factories_variable = yes
        change_variable = { 
            name = fmc_public_construction_buy_target 
            add = 10
        }
        fmc_country_update_private_public = yes
        fmc_update_construction_effect = yes
    }
}

remove_construction_scriptedgui = {

    scope = country    

    is_shown = {
        always = yes
    }
    
    ai_is_valid = {
        always = no
    }       
    is_valid = {
        var:fmc_public_construction_buy_target > 0
    }       

    effect = {          
        #custom_tooltip = "" 
        initialize_gov_factories_variable = yes
        if = {
            limit = {
                var:fmc_public_construction_buy_target > 10
            }
            change_variable = { 
                name = fmc_public_construction_buy_target 
                subtract = 10
            } 
        }
        else = {
            set_variable = { 
                name = fmc_public_construction_buy_target 
                value = 0
            }
        }
        fmc_country_update_private_public = yes
        fmc_update_construction_effect = yes
    }
}
