RP_on_weekly_pulse = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		country_update_construction_data = yes
		if = {
			limit = {
				is_ai = yes
			}
			ai_update_construction = yes
		}
	}
}

#週次パルス時の処理
##中央集権型である場合、建設データを更新
##AIである場合、建設を更新

			}

		}

	}
}
