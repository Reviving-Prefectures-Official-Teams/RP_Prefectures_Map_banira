RP_state_on_construction_start = {
	limit = {
		capital = yes
	}
	create_building = {
		building = building_general_constructor_industry
		level = 1
	}
}

#建設開始時の処理
##建設セクターを1つ作成

RP_state_on_construction_end = {
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					is_under_construction = yes
				}
			}
			any_scope_building = {
				type = bt:building_general_constructor_industry
			}
		}
		remove_building = building_general_constructor_industry
	}
}

#建設終了時の処理
##建設中の建物がない場合、建設セクターを削除
