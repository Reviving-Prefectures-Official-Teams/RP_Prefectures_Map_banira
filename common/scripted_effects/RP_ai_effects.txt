ai_update_construction = {
	set_variable = {
		name = num_gov_factories
		value = RP_ai_construction_purchase_target
	}	
}

#AIの建設購入額を更新する
##変数名をnum_gov_factories（政府工場の数）にし、
##値をRP_ai_construction_purchase_target（建設購入額の目標値）にする

RP_on_weekly_pulse = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		country_update_construction_data = yes
		if = {
			limit = {
				is_ai = yes
			}
			ai_update_construction = yes
		}
	}
}

#週次パルス時の処理
##中央集権型である場合、建設データを更新
##AIである場合、建設を更新

RP_on_border_changes = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		country_update_sectors = yes
	}
}
