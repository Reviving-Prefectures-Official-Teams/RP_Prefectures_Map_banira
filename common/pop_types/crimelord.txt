crimelord = {
	texture = "gfx/interface/icons/pops_icons/capitalists.dds"
	color = hsv{ 0.10 0.10 0.10 } #色:ダークグレー
	strata = rich #階層:リッチ
	start_quality_of_life = 30 #生活の質:20
	wage_weight = 6 #賃金の重み:6
	paid_private_wage = no #私的賃金を支払う:しない
	literacy_target = 0.70 #リテラシー目標:0.50
	dependent_wage = 0.0	# per year
	unemployment = yes #失業:する
	
	# 70% - 100% politically engaged
	political_engagement_base = 0.7 #政治的関与の基本:0.7
	political_engagement_literacy_factor = 0.3 #政治的関与のリテラシー係数:0.3
	
	political_engagement_mult = { #政治的関与の乗数
		value = 1	

		add = {
			desc = "POP_STARVATION"	
			
			if = {
				limit = { 
					standard_of_living < 20
				}
				value = 1
			}		
		}
	}	
	
	qualifications_growth_desc = "CRIMELORD_QUALIFICATIONS_DESC"
	qualifications = { #資格
		if = {
			limit = {
				is_peasant_under_serfdom = no 
				wealth < 10 #富:10
			}
			# baseline: ( wealth - 10 ) / 10 + ( literacy * 2 ), norm 2 for Wealth 20 and Literacy 50%
			add = {
				value = literacy_rate
				divide = 5
				desc = "QUALIFICATIONS_LITERACY_FACTOR"
			} #ここでは、リテラシー率を5で割り、その結果を返す
			if = {
				limit = {
					is_pop_type = laborers
				}
				multiply = {
					value = 5
					desc = "QUALIFICATIONS_FAVORED_TYPE"
				}
			} #ここでは、人口タイプが労働者の場合、5を掛けて返す
			if = {
				limit = {
					pop_acceptance >= acceptance_status_3
					# culture_accepted = no
				}
				multiply = {
					value = 5
					desc = "QUALIFICATIONS_CULTURAL_DISCRIMINATION"
				}
			} #ここでは、受け入れられた文化がない場合、0.1を掛けて返す
		}
	}
	
	portrait_age = {
		integer_range = {
			min = define:NPortrait|GRACEFUL_AGING_START
			max = define:NPortrait|GRACEFUL_AGING_END
		}
	}
	portrait_pose = { value = 0 }			
	portrait_is_female = { always = yes }
}
